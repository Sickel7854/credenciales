<script>

  const personas = {

    "001": {

      nombre: "Ana María López",

      ci: "12345678",

      cargo: "Supervisora de Producción",

      foto: "https://via.placeholder.com/120", // genérica

      telefono: "789-12345",

      estado: "válida"

    },

    "002": {

      nombre: "Jorge Ríos",

      ci: "98765432",

      cargo: "Técnico de Campo",

      foto: "https://via.placeholder.com/120", // genérica

      telefono: "765-43210",

      estado: "válida"

    }

  };


  const params = new URLSearchParams(window.location.search);

  const id = params.get("id");

  const contenedor = document.getElementById("resultado");


  if (personas[id]) {

    const p = personas[id];

    contenedor.innerHTML = `

      <img src="${p.foto}" alt="Foto de ${p.nombre}" />

      <h2>${p.nombre}</h2>

      <p><strong>CI:</strong> ${p.ci}</p>

      <p><strong>Cargo:</strong> ${p.cargo}</p>

      <p><strong>Teléfono:</strong> ${p.telefono}</p>

      <p class="valid">Credencial VÁLIDA</p>

    `;

  } else {

    contenedor.innerHTML = `

      <h2>Credencial no encontrada</h2>

      <p class="invalid">ID: ${id}</p>

    `;

  }

</script>
